Is this script safe? 
javascript: (function () {
    var currentUrl = window.location.href;
    var encodedUrl = encodeURIComponent(currentUrl);
    var archiveUrl = "https://archive.is/submit/?url=" + encodedUrl;
    window.open(archiveUrl, "_blank");
})();
